clear
heure=$(ssh lgqev13 ps -ef | grep j_tc-intergeo | grep start | awk  '{print $5}')

echo " "
echo  "Sur lgqev13 process  j_tc-intergeo start actif depuis $heure : relance O/N ?"
read rep
#/etc/init.d/j_tc-intergeo restart
[ "$rep" = O ] &&  ssh lgqev13  /etc/init.d/j_tc-intergeo restart
ssh  lgqev13 ping -c 5 www.geopostgroup.com
